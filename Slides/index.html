Vergleich zu C#

class Person {
    private readonly string name;
    private readonly int age;
    public string Name { get { return name; } }
    public string Age { get { return age; } }
    public Person(string name, int age) {
        this.name = name;
        this.age = age;
    }
}
type Person = { Name : string; Age : int }
